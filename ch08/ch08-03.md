%matplotlib widget
import matplotlib.pyplot as plt
import numpy as np

pi = np.pi

x1 = np.arange(-pi/2, pi/2, 0.01)
y1 = np.sin(x1)

x2 = np.arange(-1, 1, 0.01)
y2 = np.arcsin(x2)


fig, ax = plt.subplots()
ax.plot(x1, y1, x2, y2, 'r')
ax.axline((0,0), (1,1), color='g', linestyle='--')

ax.grid()
ax.set_aspect('equal', 'box')
ax.legend(labels=('$y = \sin(x)$', '$y = \sin^{-1}(x)$'), loc='upper left')
plt.xlim(-pi/2, pi/2)
plt.ylim(-pi/2, pi/2)
```

%matplotlib widget
import matplotlib.pyplot as plt
import numpy as np

pi = np.pi

x = np.arange(-pi/2, pi/2, 0.01)
y1 = np.tan(x)
y2 = np.arctan(x)


fig, ax = plt.subplots()
ax.plot(x, y1, x, y2, 'r')
ax.axline((0,0), (1,1), color='g', linestyle='--')

ax.grid()
ax.set_aspect('equal', 'box')
ax.legend(labels=('$y = tan(x)$', '$y = tan^{-1}(x)$'), loc='lower right')
plt.xlim(-pi/2, pi/2)
plt.ylim(-pi/2, pi/2)
```
