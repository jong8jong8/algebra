# Example 1

from sympy import symbols, Eq, linsolve

x, y, z = symbols('x, y, z') 

eq1 = Eq(x + y + z, 2)
eq2 = Eq(6*x - 4*y + 5*z, 31)
eq3 = Eq(5*x + 2*y + 2*z, 13)

linsolve([eq1, eq2, eq3], (x, y, z))
```

# Example 2

from sympy import symbols, Eq, linsolve

x, y, z = symbols('x, y, z') 

eq1 = Eq(x - 2*y + 3*z, 9)
eq2 = Eq(-x + 3*y - z, -6)
eq3 = Eq(2*x - 5*y + 5*z, 17)

linsolve([eq1, eq2, eq3], (x, y, z))

linsolve([eq1, eq2], (x, y))

linsolve([eq2, eq3], (x, y))

linsolve([eq3, eq1], (x, y))
```
