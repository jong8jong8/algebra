``` 
# Example 4

from sympy.matrices import Matrix

A = Matrix( [
    [1, 1, -1],
    [3, -2, 1],
    [1, 3, -2]
] )

A

B = Matrix( [
    [6],
    [-5],
    [14]
] )

B

X = A[:, :]
Y = A[:, :]
Z = A[:, :]

X[:, 0] = B
Y[:, 1] = B
Z[:, 2] = B

D = A.det()
Dx = X.det()
Dy = Y.det()
Dz = Z.det()

D  # check D != 0

x = Dx/D
y = Dy/D
z = Dz/D

x

y

z
```
