```
# Example 1

%matplotlib widget
import matplotlib.pyplot as plt
import numpy as np

r = 3
theta = np.pi/2

fig, ax = plt.subplots(subplot_kw={'projection': 'polar'})
ax.scatter(theta, r)
ax.set_rmax(4)
ax.set_rticks(list(np.arange(0, 5, 1)))
ax.set_rlabel_position(-22.5)
ax.grid(True)
```
