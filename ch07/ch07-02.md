```
# Example 5

from sympy import symbols, pi, sin, tan, Eq, solveset

alpha = symbols('alpha')
alpha

a, c = symbols('a, c')

alpha = 30 # degree
alpha

alpha = alpha*(pi/180) # convert degree to radian
alpha

expr_a = Eq(tan(alpha), 7/a)
expr_a

sol_a = solveset(expr_a, a)
sol_a

sol_a.evalf()

expr_c = Eq(sin(alpha), 7/c)
expr_c

sol_c = solveset(expr_c, c)
sol_c

sol_c.evalf()
```
