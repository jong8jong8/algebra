```
%matplotlib widget
import matplotlib.pyplot as plt
import numpy as np

x = np.arange(-6, 6, 0.01)
y = 2**x

x_sample = [-3, -2, -1, 0, 1, 2, 3]
y_sample = [1/8, 1/4, 1/2, 1, 2, 4, 8]

fig, ax = plt.subplots()
ax.plot(x, y)
ax.scatter(x_sample, y_sample, color='k')
ax.grid()
ax.set_aspect('equal', 'box')
plt.xlim(-6, 6)
plt.ylim(-2, 10)
```
